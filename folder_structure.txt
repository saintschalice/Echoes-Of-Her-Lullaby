# Root folder to scan
$root = "D:\EchoesOfHerLullaby\Echoes Of Her Lullaby\Assets"

# Output file
$output = Join-Path $root "FolderStructure.txt"

# Use Windows built-in TREE command
cmd /c "tree `"$root`" /a /f" | Out-File -FilePath $output -Encoding utf8

Write-Host "Folder structure saved to: $output"
